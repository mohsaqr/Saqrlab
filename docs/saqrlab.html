Installation

# From GitHub
devtools::install_github("mohsaqr/Saqrlab")
library(Saqrlab)

Requires R ≥ 4.1.0. Core imports: tna, seqHMM, dplyr, tidyr. Optional: igraph, network, parallel, future, lhs.

Saqrlab provides tools for simulating TNA data, running power analyses, and comparing networks. All simulation functions support learning state names from educational research and reproducible results via seed parameters.

Sequence Generation

simulate_sequences()

Generate Markov chain sequences from transition probabilities. Auto-generates probabilities if not provided.

simulate_sequences(n_sequences = 1000, seq_length = 20, n_states = 8, states = NULL, use_learning_states = TRUE, categories = "all", trans_matrix = NULL, init_probs = NULL, na_range = c(0, 0), include_params = FALSE, seed = NULL)
ParameterDescriptionDefault
n_sequencesNumber of sequences to generate1000
seq_lengthMaximum length of each sequence20
n_statesNumber of states8
use_learning_statesUse learning verb namesTRUE
categoriesLearning categories: "metacognitive", "cognitive", "behavioral", "social", "motivational", "affective", "group_regulation", "lms", or "all""all"
trans_matrixCustom transition matrix (rows sum to 1)NULL
init_probsCustom initial probabilitiesNULL
na_rangeRange of NAs per sequence, e.g., c(2, 5)c(0, 0)
include_paramsReturn generating parametersFALSE
seedRandom seed for reproducibilityNULL
Returns: A data.frame with sequences (rows) x time points (columns). If include_params = TRUE, returns a list with sequences, trans_matrix, and initial_probs.
Examples 
# Basic usage - auto-generates probabilities
seqs <- simulate_sequences(n_sequences = 100, seq_length = 20, n_states = 6, seed = 42)

# With specific learning category
seqs <- simulate_sequences(
  n_sequences = 100,
  n_states = 5,
  categories = "metacognitive",
  seed = 42
)

# With custom transition matrix
trans_mat <- matrix(c(0.7, 0.2, 0.1, 0.1, 0.7, 0.2, 0.2, 0.1, 0.7), nrow = 3, byrow = TRUE)
rownames(trans_mat) <- colnames(trans_mat) <- c("A", "B", "C")
init_probs <- c(A = 0.5, B = 0.3, C = 0.2)

seqs <- simulate_sequences(trans_matrix = trans_mat, init_probs = init_probs, n_sequences = 100)

# Get generating parameters back
result <- simulate_sequences(n_sequences = 100, n_states = 5, include_params = TRUE, seed = 42)
result$sequences      # data.frame
result$trans_matrix   # generating transition matrix

simulate_sequences_advanced()

Generate sequences with stable transition patterns — useful for testing edge detection in bootstrap analyses.

simulate_sequences_advanced(trans_matrix, init_probs, seq_length, n_sequences, stable_transitions, stability_prob = 0.95, unstable_mode = "random_jump", ...)
ParameterDescriptionDefault
trans_matrixSquare transition matrix (rows sum to 1)—
init_probsInitial state probabilities—
stable_transitionsList of character vectors defining stable edges, e.g., list(c("Plan", "Monitor"))—
stability_probProbability of following stable transitions (0-1)0.95
unstable_mode"random_jump", "perturb_prob", or "unlikely_jump""random_jump"
Returns: A data.frame with sequences that follow the stable transition patterns.
Examples 
# Define stable transitions (ground truth edges)
stable_transitions <- list(
  c("Plan", "Monitor"),
  c("Monitor", "Evaluate"),
  c("Evaluate", "Reflect")
)

seqs <- simulate_sequences_advanced(
  trans_matrix = trans_mat,
  init_probs = init_probs,
  seq_length = 30,
  n_sequences = 200,
  stable_transitions = stable_transitions,
  stability_prob = 0.95,
  unstable_mode = "random_jump"
)

simulate_tna_datasets()

Generate complete TNA datasets including sequences, fitted models, and generating probabilities.

simulate_tna_datasets(n_datasets = 1, n_states = 8, n_sequences = 100, seq_length = 30, model_type = "tna", use_learning_states = TRUE, categories = NULL, include_data = TRUE, include_probs = TRUE, seed = NULL, verbose = TRUE)
ParameterDescriptionDefault
n_datasetsNumber of datasets to generate1
n_statesNumber of states per network8
n_sequencesSequences per dataset100
seq_lengthMaximum sequence length30
model_type"tna", "ftna", "ctna", or "atna""tna"
use_learning_statesUse learning verb namesTRUE
include_dataInclude sequences in outputTRUE
include_probsInclude generating probabilitiesTRUE
Returns: A named list of n_datasets elements. Each contains: model (fitted tna), sequences (data.frame), transition_probs (matrix), initial_probs (vector), params (list).
Examples 
# Generate 5 complete datasets
datasets <- simulate_tna_datasets(
  n_datasets = 5,
  n_states = 6,
  n_sequences = 100,
  seed = 42
)

# Access components
datasets$dataset_1$sequences        # sequence data
datasets$dataset_1$model            # fitted TNA model
datasets$dataset_1$transition_probs # generating matrix
datasets$dataset_1$initial_probs    # initial probabilities
datasets$dataset_1$params$state_names  # state names used

# Generate fTNA datasets
ftna_datasets <- simulate_tna_datasets(
  n_datasets = 3,
  model_type = "ftna",
  seed = 42
)
